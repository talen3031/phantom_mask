22#1.查詢依指定時間與星期幾列出有營業的藥局 

23@pharmacy_bp.route('/pharmacies/open', methods=['GET']) 

24def list_open_pharmacies(): 

25 

26 day = request.args.get('day') # 例如 'Mon' 

27 time = request.args.get('time') # 例如 '15:30' 

28 

29 valid_days = ['Mon', 'Tue', 'Wed', 'Thur', 'Fri', 'Sat', 'Sun'] 

30 time_pattern = re.compile(r'^\d{2}:\d{2}$') # 格式必須為兩位數:兩位數 

31 

32 # 檢查參數是否存在 

33 if not day or not time: 

34 return jsonify({"error": "please enter day, time"}), 400 

35 

36 # 檢查 day 是否正確 

37 if day not in valid_days: 

38 return jsonify({"error": "please enter day in correct format(ex.Mon)"}), 400 

39 

40 # 檢查 time 格式 

41 if not time_pattern.match(time): 

42 return jsonify({"error": "please enter time in correct format(00:00~23:59) (ex.14:30)"}), 400 

43 

44 # 進階：檢查 HH 和 MM 合理範圍 

45 hour, minute = map(int, time.split(':')) 

46 if hour < 0 or hour > 23 or minute < 0 or minute > 59: 

47 return jsonify({"error": "please enter time in correct format (00:00~23:59) (ex.14:30) "}), 400 

48 

49 

50 conn = get_db() 

51 cur = conn.cursor() 

52 # 開始查詢 

53 query = """ 

54 SELECT DISTINCT p.* 

55 FROM pharmacy p 

56 JOIN opening_hour o ON p.id = o.pharmacy_id 

57 WHERE o.day_of_week = ? 

58 AND o.open_time <= ? 

59 AND o.close_time >= ? 

60 """ 

61 cur.execute(query, (day, time, time)) 

62 rows = cur.fetchall() 

63 conn.close()

81 

82 # 驗證數值格式 

83 try: 

84 min_price = float(min_price) 

85 max_price = float(max_price) 

86 count = int(count) 

87 except: 

88 return jsonify({"error": "please enter min_price, max_price, count in correct way"}), 400 

89 

90 # 支援的 op 

91 op_map = { 

92 'gt': '>', 

93 'lt': '<', 

94 'ge': '>=', 

95 'le': '<=', 

96 'eq': '=' 

97 } 

98 if op not in op_map: 

99 return jsonify({"error": "op must be gt, lt, ge, le, eq , one of them."}), 400 

100 

101 having_sql = f'mask_count {op_map[op]} ?' 

102 

103 query = f''' 

104 SELECT 

105 pharmacy.id, 

106 pharmacy.name, 

107 pharmacy.cash_balance, 

108 COUNT(mask.id) as mask_count 

109 FROM 

110 pharmacy 

111 JOIN 

112 mask ON pharmacy.id = mask.pharmacy_id 

113 WHERE 

114 mask.price BETWEEN ? AND ? 

115 GROUP BY 

116 pharmacy.id 

117 HAVING 

118 {having_sql} 

119 ''' 

120 conn = get_db()

122 cur.execute(query, (min_price, max_price, count)) 

123 rows = cur.fetchall() 

124 conn.close()
